# 简单实现
```python
'''
https://www.lanqiao.cn/problems/303/learning/?page=1&first_category_id=1&second_category_id=6&name=%E8%83%8C%E5%8C%85
'''
import os
import sys

# 请在此输入您的代码
t, n = map(int, input().split())
weights = []
values = []
for _ in range(n):
    w, v = map(int, input().split())
    weights.append(w)
    values.append(v)

# dp[i][w]表示前 i 件物品，放入容量不超过 w 的背包时可获得的最大价值。
dp = [[0] * (t + 1) for _ in range(n + 1)]

# 遍历每个物品
for i in range(1, n + 1):
    # 遍历每一种可能的背包容量
    for w in range(t + 1):
        if w < weights[i - 1]:  # 如果 当前容量小于当前物品，则继承上一个状态
            dp[i][w] = dp[i - 1][w]
        else:
            # 当前物品可选，取放与不放的最大值
            dp[i][w] = max(dp[i-1][w], dp[i-1][w-weights[i-1]] + values[i-1])

print(dp[n][t])  # 前n个物品，容量为t时的最大价值



```
# 滚动数组实现
```python
import os
import sys

# 请在此输入您的代码
t, n = map(int, input().split())
weights = []
values = []
for _ in range(n):
    w, v = map(int, input().split())
    weights.append(w)
    values.append(v)

# dp[i][w]表示前 i 件物品，放入容量不超过 w 的背包时可获得的最大价值。
dp = [0] * (t + 1)

# 遍历每个物品
for i in range(n):
    # 遍历每一种可能的背包容量
    for w in range(t, weights[i] - 1, -1):
        dp[w] = max(dp[w], dp[w - weights[i]] + values[i])

print(dp[t])
```
